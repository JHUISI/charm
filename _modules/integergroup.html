[docs]    def paramgen(self, secparam):
        while True:
           p, q = randomPrime(secparam), randomPrime(secparam)
           if isPrime(p) and isPrime(q) and gcd(p * q, (p - 1) * (q - 1)) == 1:
              break
        n = p * q
        self.p = p
        self.q = q
        self.n = n
        return (p, q, n)
